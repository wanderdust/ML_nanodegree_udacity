Your algorithm for TD prediction has five arguments:

  • num_episodes: This is the number of episodes that are generated through agent-environment interaction.

from collections import defaultdict, deque
import sys

def td_prediction(env, num_episodes, policy, alpha, gamma=1.0):
    # initialize empty dictionaries of floats
    V = defaultdict(float)
    # loop over episodes
    for i_episode in range(1, num_episodes+1):
        # monitor progress
        if i_episode % 100 == 0:
            print("\rEpisode {}/{}".format(i_episode, num_episodes), end="")
            sys.stdout.flush()
        
        ## TODO: complete the function
        state = env.reset()
        
        while True:
            action = policy[state]
            next_state, reward, done, info = env.step(action)
            V[state] = V[state] + alpha * (reward + gamma*V[next_state] - V[state])
            state = next_state
            
            if done:
                break
    return V

def choose_action(env, q, epsilon=0.01):
    nA = env.action_space.n
    
    prob_non_greedy = (epsilon/nA)
    prob_greedy = 1-epsilon + epsilon/nA
    
    probs = np.full(4, prob_non_greedy)
    probs[np.argmax(q)] = prob_greedy
    
    action = np.random.choice(np.arange(nA), p=probs)
    
    return action

def sarsa(env, num_episodes, alpha, gamma=1.0):
    # initialize action-value function (empty dictionary of arrays)
    Q = defaultdict(lambda: np.zeros(env.nA))
    # initialize performance monitor
    # loop over episodes
    for i_episode in range(1, num_episodes+1):
        # monitor progress
        if i_episode % 100 == 0:
            print("\rEpisode {}/{}".format(i_episode, num_episodes), end="")
            sys.stdout.flush()   
        
        ## TODO: complete the function
        epsilon = 1/i_episode # Decaying epsilon
        state = env.reset()
        action = choose_action(env, Q[state], epsilon)
        
        while True:
            next_state, reward, done, info = env.step(action)
            next_action = choose_action(env, Q[next_state], epsilon)
            Q[state][action] = Q[state][action] + alpha*(reward + gamma*Q[next_state][next_action] - Q[state][action])
            action, state = next_action, next_state
            
            if done:
                Q[state][action] = Q[state][action] + alpha*(reward + gamma*0 - Q[state][action])
                break
        
    return Q

def expected_sarsa(env, num_episodes, alpha, gamma=1.0):
    # initialize empty dictionary of arrays
    Q = defaultdict(lambda: np.zeros(env.nA))
    # loop over episodes
    for i_episode in range(1, num_episodes+1):
        # monitor progress
        if i_episode % 100 == 0:
            print("\rEpisode {}/{}".format(i_episode, num_episodes), end="")
            sys.stdout.flush()
        
        ## TODO: complete the function
        epsilon = 1/i_episode
        state = env.reset()
        
        while True:
            action = choose_action(env, Q[state], epsilon)
            next_state, reward, done, info = env.step(action)
            
            probs = get_action_probs(env, Q[next_state])
            expected = sum([q_a*probs[i] for i, q_a in enumerate(Q[next_state])])
            
            Q[state][action] = Q[state][action] + alpha*(reward + gamma*expected - Q[state][action])
            state = next_state
            
            if done:
                Q[state][action] = Q[state][action] + alpha*(reward + gamma*0 - Q[state][action])
                break
        
    return Q
